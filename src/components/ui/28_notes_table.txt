create extension if not exists pgcrypto;

create table if not exists public.show_notes (
  id uuid primary key default gen_random_uuid(),
  organization_id uuid not null,
  show_id uuid not null references public.shows(id) on delete cascade,
  title text not null default 'Untitled Note',
  body_html text not null default '',
  sort_order integer not null default 0,
  created_by_membership_id uuid null,
  updated_by_membership_id uuid null,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create index if not exists show_notes_show_id_idx
  on public.show_notes (show_id);

create index if not exists show_notes_org_show_idx
  on public.show_notes (organization_id, show_id);

create or replace function public.set_show_notes_updated_at()
returns trigger
language plpgsql
as $$
begin
  new.updated_at = now();
  return new;
end;
$$;

drop trigger if exists trg_show_notes_updated_at on public.show_notes;

create trigger trg_show_notes_updated_at
before update on public.show_notes
for each row
execute function public.set_show_notes_updated_at();

alter table public.show_notes enable row level security;

drop policy if exists "show_notes_select_org" on public.show_notes;
drop policy if exists "show_notes_insert_org" on public.show_notes;
drop policy if exists "show_notes_update_org" on public.show_notes;
drop policy if exists "show_notes_delete_org" on public.show_notes;

create policy "show_notes_select_org"
on public.show_notes
for select
to authenticated
using (
  exists (
    select 1
    from public.organization_memberships om
    join public.shows s
      on s.organization_id = om.organization_id
    where om.user_id = auth.uid()
      and om.status = 'active'
      and s.id = show_notes.show_id
      and s.organization_id = show_notes.organization_id
  )
);

create policy "show_notes_insert_org"
on public.show_notes
